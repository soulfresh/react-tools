## Example Usage

Here's an example of how you can use this to create your
own `<Action>` component.

```jsx
import { combineClasses } from '@thesoulfresh/utils';
import { UnstyledAction } from '@thesoulfresh/react-tools';

export const Action = React.forwardRef(({
  button = false,
  transparent = false,
  pill = false,
  className,
  ...rest
}, ref) => {
  return (
    <UnstyledAction
      className={combineClasses(
        // Base styles for your buttons and links
        styles.Action,
        // Ensure outer styles are also applied
        className,
        // Apply your custom button or link styles
        // depending on what is to be rendered.
        button ? styles.button : styles.link,
        // Render the pill version of your button.
        pill ? styles.pill : null,
        // Render a button with a transparent background.
        transparent ? styles.transparent : null,
      )}
      {...rest}
    />
  );
});

// Elsewhere you can use this to render links by using:
<Action href="http://www.google.com" blank>My Link</Action>

// Or you can render buttons using:
<Action button pill onClick={() => alert('button clicked!')}>
  My Button
</Action>
```
