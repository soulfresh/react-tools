#### Controlled

You can control the open/closed state yourself by passing the `isOpen` prop.

<Canvas>
  <Story name="Controlled">
    <Column>
    {() => {
      const [isOpen, setIsOpen] = React.useState(false);
      return (
        <ExampleDropdown
          isOpen={isOpen}
          onOpen={() => setIsOpen(true)}
          onClose={() => setIsOpen(false)}
          content={(first, last) =>
            <div>
              <div>ðŸ‘‹ ðŸŒŽ</div>
              <div><button ref={first}>First Button</button></div>
              <div><button ref={last}>Last Button</button></div>
            </div>
          }
        >
          <button>Menu</button>
        </ExampleDropdown>
      );
    }}
    </Column>
  </Story>
</Canvas>
